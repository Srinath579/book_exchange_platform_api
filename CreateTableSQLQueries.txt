//Command to Create Users Table
CREATE TABLE USERS (
	USER_ID INT PRIMARY KEY AUTO_INCREMENT,
	USER_NAME VARCHAR(20),
    PASSWORD VARCHAR(50),
    EMAIL_ID VARCHAR(50) UNIQUE,
    REG_DATE DATETIME
);

//Commmand to Create Books Table
CREATE TABLE BOOKS (
        BOOK_ID INT PRIMARY KEY AUTO_INCREMENT,
        TITLE VARCHAR(255),
        AUTHOR VARCHAR(255),
        GENRE VARCHAR(255),
        BOOK_CONDITION ENUM('G', 'SD', 'D'),
        AVAILABILITY_STATUS ENUM('AVAILABLE', 'UNAVAILABLE', 'OCCUPIED'),
        BOOK_IMAGE BLOB,
        OWNER_ID INT,
        OWNER_EMAIL_ID VARCHAR(50),
        OWNER_NAME VARCHAR(20),
        FOREIGN KEY (OWNER_ID) REFERENCES USERS(USER_ID),
	    FOREIGN KEY (OWNER_EMAIL_ID) REFERENCES USERS(EMAIL_ID)
);

//Command to Create Exchange Table
CREATE TABLE EXCHANGE (
        EXCHANGE_ID INT PRIMARY KEY AUTO_INCREMENT,
        REQUESTER_EMAIL_ID VARCHAR(50),
        REQUESTER_NAME VARCHAR(20),
        OWNER_EMAIL_ID VARCHAR(50),
        BOOK_ID INT,
        BOOK_NAME VARCHAR(255),
        DELIVERY_METHOD ENUM('Postal', 'Nodal Point', 'Home Pickup'),
        DURATION_OF_BOOK_1 INT,
        DURATION_OF_BOOK_2 ENUM('Days', 'Weeks', 'Months', 'Years'),
        STATUS ENUM('Requested', 'Accepted', 'Declined'),
        FOREIGN KEY (REQUESTER_EMAIL_ID) REFERENCES USERS(EMAIL_ID),
        FOREIGN KEY (OWNER_EMAIL_ID) REFERENCES USERS(EMAIL_ID)
);
